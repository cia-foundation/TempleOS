#define TK_NUM		1
#define TK_OP		2

I64 Lex(U8 **_src,I64 *num)
{ //See $LK,"Lex",A="MN:Lex"$().
  U8 *src=*_src;
  I64 i;
  while (TRUE) {
    switch (*src) {
      case 0:
      case ';':
	*_src=src;
	return TK_EOF;
      case CH_SPACE:
      case '\r':
      case '\n':
	src++;
	break;
      case '0'...'9':
	i=0;
	do {
	  i=i*10+*src-'0';
	  src++;
	} while ('0'<=*src<='9');
	*num=i;
	*_src=src;
	return TK_NUM;
      case '*':
	*num=OP_MUL;
	*_src=src+1;
	return TK_OP;
      case '/':
	*num=OP_DIV;
	*_src=src+1;
	return TK_OP;
      case '+':
	*num=OP_ADD;
	*_src=src+1;
	return TK_OP;
      case '-':
	*num=OP_SUB;
	*_src=src+1;
	return TK_OP;
      case '(':
	*_src=src+1;
	return TK_LEFT;
      case ')':
	*_src=src+1;
	return TK_RIGHT;
      default:
	throw;
    }
  }
}
